149  private void setupBridge(final StartTime startTime) {
150    // Signal to the clr buffered log handler that the driver has started and that
151    // we can begin logging
152    LOG.log(Level.INFO, "Initializing CLRBufferedLogHandler...");
153    try (final LoggingScope lb = this.loggingScopeFactory.setupBridge()) {
154      final CLRBufferedLogHandler handler = getCLRBufferedLogHandler();
155      if (handler == null) {
156        LOG.log(Level.WARNING, "CLRBufferedLogHandler could not be initialized");
157      } else {
158        handler.setDriverInitialized();
159        LOG.log(Level.INFO, "CLRBufferedLogHandler init complete.");
160      }
161
162      LOG.log(Level.INFO, "StartTime: {0}", new Object[]{startTime});
163      String portNumber = httpServer == null ? null : Integer.toString((httpServer.getPort()));
164      long[] handlers = NativeInterop.CallClrSystemOnStartHandler(startTime.toString(), portNumber);
165      if (handlers != null) {
166        if (handlers.length != NativeInterop.nHandlers) {
167          throw new RuntimeException(
168              String.format("%s handlers initialized in CLR while native bridge is expecting %s handlers",
169                  String.valueOf(handlers.length),
170                  String.valueOf(NativeInterop.nHandlers)));
171        }
172        this.evaluatorRequestorHandler = handlers[NativeInterop.Handlers.get(NativeInterop.EvaluatorRequestorKey)];
173        this.allocatedEvaluatorHandler = handlers[NativeInterop.Handlers.get(NativeInterop.AllocatedEvaluatorKey)];
174        this.activeContextHandler = handlers[NativeInterop.Handlers.get(NativeInterop.ActiveContextKey)];
175        this.taskMessageHandler = handlers[NativeInterop.Handlers.get(NativeInterop.TaskMessageKey)];
176        this.failedTaskHandler = handlers[NativeInterop.Handlers.get(NativeInterop.FailedTaskKey)];
177        this.failedEvaluatorHandler = handlers[NativeInterop.Handlers.get(NativeInterop.FailedEvaluatorKey)];
178        this.httpServerEventHandler = handlers[NativeInterop.Handlers.get(NativeInterop.HttpServerKey)];
179        this.completedTaskHandler = handlers[NativeInterop.Handlers.get(NativeInterop.CompletedTaskKey)];
180        this.runningTaskHandler = handlers[NativeInterop.Handlers.get(NativeInterop.RunningTaskKey)];
181        this.suspendedTaskHandler = handlers[NativeInterop.Handlers.get(NativeInterop.SuspendedTaskKey)];
182        this.completedEvaluatorHandler = handlers[NativeInterop.Handlers.get(NativeInterop.CompletedEvaluatorKey)];
183        this.closedContextHandler = handlers[NativeInterop.Handlers.get(NativeInterop.ClosedContextKey)];
184        this.failedContextHandler = handlers[NativeInterop.Handlers.get(NativeInterop.FailedContextKey)];
185        this.contextMessageHandler = handlers[NativeInterop.Handlers.get(NativeInterop.ContextMessageKey)];
186        this.driverRestartHandler = handlers[NativeInterop.Handlers.get(NativeInterop.DriverRestartKey)];
187        this.driverRestartActiveContextHandler = handlers[NativeInterop.Handlers.get(NativeInterop.DriverRestartActiveContextKey)];
188        this.driverRestartRunningTaskHandler = handlers[NativeInterop.Handlers.get(NativeInterop.DriverRestartRunningTaskKey)];
189      }
190
191      try (final LoggingScope lp = this.loggingScopeFactory.getNewLoggingScope("setupBridge::ClrSystemHttpServerHandlerOnNext")) {
192        final HttpServerEventBridge httpServerEventBridge = new HttpServerEventBridge("SPEC");
193        NativeInterop.ClrSystemHttpServerHandlerOnNext(this.httpServerEventHandler, httpServerEventBridge, this.interopLogger);
194        final String specList = httpServerEventBridge.getUriSpecification();
195        LOG.log(Level.INFO, "Starting http server, getUriSpecification: {0}", specList);
196        if (specList != null) {
197          final String[] specs = specList.split(":");
198          for (final String s : specs) {
199            final HttpHandler h = new HttpServerBridgeEventHandler();
200            h.setUriSpecification(s);
201            this.httpServer.addHttpHandler(h);
202          }
203        }
204      }
205      this.clrBridgeSetup = true;
206    }
207    LOG.log(Level.INFO, "CLR Bridge setup.");
208  }

210  private CLRBufferedLogHandler getCLRBufferedLogHandler() {
211    for (Handler handler : Logger.getLogger("").getHandlers()) {
212      if (handler instanceof CLRBufferedLogHandler)
213        return (CLRBufferedLogHandler) handler;
214    }
215    return null;

218  private void submitEvaluator(final AllocatedEvaluator eval, EvaluatorType type) {
219    synchronized (JobDriver.this) {
220      eval.setType(type);
221      LOG.log(Level.INFO, "Allocated Evaluator: {0}, total running running {1}",
222          new Object[]{eval.getId(), JobDriver.this.contexts.size()});
223      if (JobDriver.this.allocatedEvaluatorHandler == 0) {
224        throw new RuntimeException("Allocated Evaluator Handler not initialized by CLR.");
225      }
226      AllocatedEvaluatorBridge allocatedEvaluatorBridge = new AllocatedEvaluatorBridge(eval, JobDriver.this.nameServerInfo);
227      NativeInterop.ClrSystemAllocatedEvaluatorHandlerOnNext(JobDriver.this.allocatedEvaluatorHandler, allocatedEvaluatorBridge, this.interopLogger);
228    }
229  }

231  /**
232   * Submit a Task to a single Evaluator.
233   */
234  private void submit(final ActiveContext context) {
235    try {
236      LOG.log(Level.INFO, "Send task to context: {0}", new Object[]{context});
237      if (JobDriver.this.activeContextHandler == 0) {
238        throw new RuntimeException("Active Context Handler not initialized by CLR.");
239      }
240      ActiveContextBridge activeContextBridge = new ActiveContextBridge(context);
241      NativeInterop.ClrSystemActiveContextHandlerOnNext(JobDriver.this.activeContextHandler, activeContextBridge, JobDriver.this.interopLogger);
242    } catch (final Exception ex) {
243      LOG.log(Level.SEVERE, "Fail to submit task to active context");
244      context.close();
245      throw new RuntimeException(ex);
246    }
247  }

283   */
284  final class CompletedTaskHandler implements EventHandler<CompletedTask> {
285    @Override
286    public void onNext(final CompletedTask task) {
287      LOG.log(Level.INFO, "Completed task: {0}", task.getId());
288      try (final LoggingScope ls = loggingScopeFactory.taskCompleted(task.getId())) {
289        // Take the message returned by the task and add it to the running result.
290        String result = "default result";
291        try {
292          result = new String(task.get());
293        } catch (final Exception e) {
294          LOG.log(Level.WARNING, "failed to decode task outcome");
295        }
296        LOG.log(Level.INFO, "Return results to the client:\n{0}", result);
297        JobDriver.this.jobMessageObserver.sendMessageToClient(JVM_CODEC.encode(result));
298        if (JobDriver.this.completedTaskHandler == 0) {
299          LOG.log(Level.INFO, "No CLR handler bound to handle completed task.");
300        } else {
301          LOG.log(Level.INFO, "CLR CompletedTaskHandler handler set, handling things with CLR handler.");
302          CompletedTaskBridge completedTaskBridge = new CompletedTaskBridge(task);
303          NativeInterop.ClrSystemCompletedTaskHandlerOnNext(JobDriver.this.completedTaskHandler, completedTaskBridge, JobDriver.this.interopLogger);
304        }
305      }
306    }

309  /**
310   * Receive notification that the entire Evaluator had failed.
311   */
312  final class FailedEvaluatorHandler implements EventHandler<FailedEvaluator> {
313    @Override
314    public void onNext(final FailedEvaluator eval) {
315      try (final LoggingScope ls = loggingScopeFactory.evaluatorFailed(eval.getId())) {
316        synchronized (JobDriver.this) {
317          LOG.log(Level.SEVERE, "FailedEvaluator", eval);
318          for (final FailedContext failedContext : eval.getFailedContextList()) {
319            String failedContextId = failedContext.getId();
320            LOG.log(Level.INFO, "removing context " + failedContextId + " from job driver contexts.");
321            JobDriver.this.contexts.remove(failedContextId);
322          }
323          String message = "Evaluator " + eval.getId() + " failed with message: "
324              + eval.getEvaluatorException().getMessage();
325          JobDriver.this.jobMessageObserver.sendMessageToClient(message.getBytes());
326
327          if (failedEvaluatorHandler == 0) {
328            if (JobDriver.this.clrBridgeSetup) {
329              message = "No CLR FailedEvaluator handler was set, exiting now";
330              LOG.log(Level.WARNING, message);
331              JobDriver.this.jobMessageObserver.sendMessageToClient(message.getBytes());
332              return;
333            } else {
334              clock.scheduleAlarm(0, new EventHandler<Alarm>() {
335                @Override
336                public void onNext(final Alarm time) {
337                  if (JobDriver.this.clrBridgeSetup) {
338                    handleFailedEvaluatorInCLR(eval);
339                  } else {
340                    LOG.log(Level.INFO, "Waiting for CLR bridge to be set up");
341                    clock.scheduleAlarm(5000, this);
342                  }
343                }
344              });
345            }
346          } else {
347            handleFailedEvaluatorInCLR(eval);
348          }
349        }
350      }
351    }
352
353    private void handleFailedEvaluatorInCLR(final FailedEvaluator eval) {
354      final String message = "CLR FailedEvaluator handler set, handling things with CLR handler.";
355      LOG.log(Level.INFO, message);
356      FailedEvaluatorBridge failedEvaluatorBridge = new FailedEvaluatorBridge(eval, JobDriver.this.evaluatorRequestor, JobDriver.this.isRestarted, loggingScopeFactory);
357      NativeInterop.ClrSystemFailedEvaluatorHandlerOnNext(JobDriver.this.failedEvaluatorHandler, failedEvaluatorBridge, JobDriver.this.interopLogger);
358      int additionalRequestedEvaluatorNumber = failedEvaluatorBridge.getNewlyRequestedEvaluatorNumber();
359      if (additionalRequestedEvaluatorNumber > 0) {
360        LOG.log(Level.INFO, "number of additional evaluators requested after evaluator failure: " + additionalRequestedEvaluatorNumber);
361      }
362      JobDriver.this.jobMessageObserver.sendMessageToClient(message.getBytes());
363    }
364  }

366  final class HttpServerBridgeEventHandler implements HttpHandler {
367    private String uriSpecification;
368
369    /**
370     * returns URI specification for the handler
371     */
372    @Override
373    public String getUriSpecification() {
374      return uriSpecification;
375    }
376
377    public void setUriSpecification(String s) {
378      uriSpecification = s;
379    }
380
381    /**
382     * process http request
383     */
384    @Override
385    public void onHttpRequest(final ParsedHttpRequest parsedHttpRequest, final HttpServletResponse response) throws IOException, ServletException {
386      LOG.log(Level.INFO, "HttpServerBridgeEventHandler onHttpRequest: {0}", parsedHttpRequest.getRequestUri());
387      try (final LoggingScope ls = loggingScopeFactory.httpRequest(parsedHttpRequest.getRequestUri())) {
388        final AvroHttpSerializer httpSerializer = new AvroHttpSerializer();
389        final AvroHttpRequest avroHttpRequest = httpSerializer.toAvro(parsedHttpRequest);
390        final byte[] requestBytes = httpSerializer.toBytes(avroHttpRequest);
391
392        try {
393          final HttpServerEventBridge httpServerEventBridge = new HttpServerEventBridge(requestBytes);
394          NativeInterop.ClrSystemHttpServerHandlerOnNext(JobDriver.this.httpServerEventHandler, httpServerEventBridge, JobDriver.this.interopLogger);
395          final String responseBody = new String(httpServerEventBridge.getQueryResponseData(), "UTF-8");
396          response.getWriter().println(responseBody);
397          LOG.log(Level.INFO, "HttpServerBridgeEventHandler onHttpRequest received response: {0}", responseBody);
398        } catch (final Exception ex) {
399          LOG.log(Level.SEVERE, "Fail to invoke CLR Http Server handler", ex);
400          throw new RuntimeException(ex);
401        }
402      }
403    }
404  }

453  final class DriverRestartRunningTaskHandler implements EventHandler<RunningTask> {
454    @Override
455    public void onNext(final RunningTask task) {
456      try (final LoggingScope ls = loggingScopeFactory.driverRestartRunningTask(task.getId())) {
457        clock.scheduleAlarm(0, new EventHandler<Alarm>() {
458          @Override
459          public void onNext(final Alarm time) {
460            if (JobDriver.this.clrBridgeSetup) {
461              if (JobDriver.this.driverRestartRunningTaskHandler != 0) {
462                LOG.log(Level.INFO, "CLR driver restart RunningTask handler implemented, now handle it in CLR.");
463                NativeInterop.ClrSystemDriverRestartRunningTaskHandlerOnNext(JobDriver.this.driverRestartRunningTaskHandler, new RunningTaskBridge(task));
464              } else {
465                LOG.log(Level.WARNING, "No CLR driver restart RunningTask handler implemented, done with DriverRestartRunningTaskHandler.");
466              }
467            } else {
468              LOG.log(Level.INFO, "Waiting for driver to complete restart process before checking out CLR driver restart RunningTaskHandler...");
469              clock.scheduleAlarm(2000, this);
470            }
471          }
472        });
473      }
474    }

509  final class StartHandler implements EventHandler<StartTime> {
510    @Override
511    public void onNext(final StartTime startTime) {

516

523          NativeInterop.ClrSystemEvaluatorRequstorHandlerOnNext(JobDriver.this.evaluatorRequestorHandler, evaluatorRequestorBridge, JobDriver.this.interopLogger);
524          // get the evaluator numbers set by CLR handler
525          LOG.log(Level.INFO, "evaluator requested at start up: " + evaluatorRequestorBridge.getEvaluatorNumber());
526        }
527      }
528    }

669      LOG.log(Level.SEVERE, "FailedContext", context);
670      try (final LoggingScope ls = loggingScopeFactory.evaluatorFailed(context.getId())) {
671        if (JobDriver.this.failedContextHandler != 0) {
672          FailedContextBridge failedContextBridge = new FailedContextBridge(context);
673          // if CLR implements the failed context handler, handle it in CLR
674          LOG.log(Level.INFO, "Handling the event of failed context in CLR bridge.");
675          NativeInterop.ClrSystemFailedContextHandlerOnNext(JobDriver.this.failedContextHandler, failedContextBridge);
676        }
677        synchronized (JobDriver.this) {
678          JobDriver.this.contexts.remove(context.getId());
679        }
680        Optional<byte[]> err = context.getData();
681        if (err.isPresent()) {
682          JobDriver.this.jobMessageObserver.sendMessageToClient(err.get());
683        }
684      }
685    }
