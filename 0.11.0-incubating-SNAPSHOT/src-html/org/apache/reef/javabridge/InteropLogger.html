019package org.apache.reef.javabridge;
020
021import java.util.HashMap;
022import java.util.logging.Level;
023import java.util.logging.Logger;
024
025public class InteropLogger {
026  private static final Logger LOG = Logger.getLogger("InteropLogger");
027  HashMap<Integer, Level> levelHashMap;
028
029  {
030    levelHashMap = new HashMap<Integer, Level>();
031    levelHashMap.put(Level.OFF.intValue(), Level.OFF);
032    levelHashMap.put(Level.SEVERE.intValue(), Level.SEVERE);
033    levelHashMap.put(Level.WARNING.intValue(), Level.WARNING);
034    levelHashMap.put(Level.INFO.intValue(), Level.INFO);
035
036    levelHashMap.put(Level.CONFIG.intValue(), Level.CONFIG);
037    levelHashMap.put(Level.FINE.intValue(), Level.FINE);
038    levelHashMap.put(Level.FINER.intValue(), Level.FINER);
039
040    levelHashMap.put(Level.FINEST.intValue(), Level.FINEST);
041    levelHashMap.put(Level.ALL.intValue(), Level.ALL);
042  }
043
044  public void Log(int intLevel, String message) {
045    if (levelHashMap.containsKey(intLevel)) {
046      Level level = levelHashMap.get(intLevel);
047      LOG.log(level, message);
048    } else {
049
050      LOG.log(Level.WARNING, "Level " + intLevel + " is not a valid Log level");
051      LOG.log(Level.WARNING, message);
052    }
053
054  }
055}
