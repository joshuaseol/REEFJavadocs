001package src;
002
003import org.apache.reef.driver.evaluator.AllocatedEvaluator;
004import org.apache.reef.driver.evaluator.EvaluatorRequest;
005import org.apache.reef.driver.evaluator.EvaluatorRequestor;
006import org.apache.reef.driver.task.TaskConfiguration;
007import org.apache.reef.tang.Configuration;
008import org.apache.reef.tang.annotations.Unit;
009import org.apache.reef.wake.EventHandler;
010import org.apache.reef.wake.time.event.StartTime;
011
012import javax.inject.Inject;
013import java.util.logging.Level;
014import java.util.logging.Logger;
015
016/**
017 * The Driver code for the Hello REEF Application
018 */
019@Unit
020public final class TestDriver {
021
022  private static final Logger LOG = Logger.getLogger(TestDriver.class.getName());
023
024  private final EvaluatorRequestor requestor;
025
026  /**
027   * Job driver constructor - instantiated via TANG.
028   *
029   * @param requestor evaluator requestor object used to create new evaluator containers.
030   */
031  @Inject
032  public TestDriver(final EvaluatorRequestor requestor) {
033    this.requestor = requestor;
034    LOG.log(Level.FINE, "Instantiated 'HelloDriver'");
035  }
036
037  /**
038   * Handles the StartTime event: Request as single Evaluator.
039   */
040  public final class StartHandler implements EventHandler<StartTime> {
041    @Override
042    public void onNext(final StartTime startTime) {
043        TestDriver.this.requestor.submit(EvaluatorRequest.newBuilder()
044          .setNumber(1)
045          .setMemory(64)
046          .setNumberOfCores(1)
047          .build());
048      LOG.log(Level.INFO, "Requested Evaluator.");
049    }
050  }
051
052  /**
053   * Handles AllocatedEvaluator: Submit the HelloTask
054   */
055  public final class EvaluatorAllocatedHandler implements EventHandler<AllocatedEvaluator> {
056    @Override
057    public void onNext(final AllocatedEvaluator allocatedEvaluator) {
058      LOG.log(Level.INFO, "Submitting TestREEF task to AllocatedEvaluator: {0}", allocatedEvaluator);
059      final Configuration taskConfiguration = TaskConfiguration.CONF
060          .set(TaskConfiguration.IDENTIFIER, "HelloREEFTask")
061          .set(TaskConfiguration.TASK, TestTask.class)
062          .build();
063      allocatedEvaluator.submitTask(taskConfiguration);
064    }
065  }
066}
