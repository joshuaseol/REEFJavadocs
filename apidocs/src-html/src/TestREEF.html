001package src;
002
003import org.apache.reef.client.DriverConfiguration;
004import org.apache.reef.client.DriverLauncher;
005import org.apache.reef.client.LauncherStatus;
006import org.apache.reef.runtime.local.client.LocalRuntimeConfiguration;
007import org.apache.reef.tang.Configuration;
008import org.apache.reef.tang.exceptions.BindException;
009import org.apache.reef.tang.exceptions.InjectionException;
010import org.apache.reef.util.EnvironmentUtils;
011
012import java.util.logging.Level;
013import java.util.logging.Logger;
014
015/**
016 * The Client for TestREEF example.
017 */
018public final class TestREEF {
019
020  private static final Logger LOG = Logger.getLogger(TestREEF.class.getName());
021
022  /**
023   * Number of milliseconds to wait for the job to complete.
024   */
025  private static final int JOB_TIMEOUT = 1000000; // 1000 sec.
026
027  /**
028   * @return the configuration of the TestREEF driver.
029   */
030  public static Configuration getDriverConfiguration() {
031    return DriverConfiguration.CONF
032        .set(DriverConfiguration.GLOBAL_LIBRARIES, EnvironmentUtils.getClassLocation(TestDriver.class))
033        .set(DriverConfiguration.DRIVER_IDENTIFIER, "TestREEF")
034        .set(DriverConfiguration.ON_DRIVER_STARTED, TestDriver.StartHandler.class)
035        .set(DriverConfiguration.ON_EVALUATOR_ALLOCATED, TestDriver.EvaluatorAllocatedHandler.class)
036        .build();
037  }
038
039  public static LauncherStatus runTestReef(final Configuration runtimeConf, final int timeOut)
040      throws BindException, InjectionException {
041    final Configuration driverConf = getDriverConfiguration();
042    return DriverLauncher.getLauncher(runtimeConf).run(driverConf, timeOut);
043  }
044
045  /**
046   * Start TestREEF job. Runs method runTesteef().
047   *
048   * @param args command line parameters.
049   * @throws BindException      configuration error.
050   * @throws InjectionException configuration error.
051   */
052  public static void main(final String[] args) throws BindException, InjectionException {
053    final Configuration runtimeConfiguration = LocalRuntimeConfiguration.CONF
054        .set(LocalRuntimeConfiguration.NUMBER_OF_THREADS, 2)
055        .build();
056    final LauncherStatus status = runTestReef(runtimeConfiguration, JOB_TIMEOUT);
057    LOG.log(Level.INFO, "REEF job completed: {0}", status);
058  }
059
060}
