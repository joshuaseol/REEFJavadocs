001package src;
002
003import java.sql.DriverManager;
004import java.sql.ResultSet;
005import java.sql.ResultSetMetaData;
006import java.sql.SQLException;
007import java.util.Scanner;
008
009public class DatabaseClient {
010
011   public static void main(String[] args) throws Exception {
012           //Load Driver
013           try {
014                    System.out.println("Loading driver...");
015                    Class.forName("com.mysql.jdbc.Driver");
016                    System.out.println("Driver loaded!");
017                } catch (ClassNotFoundException e) {
018                    throw new RuntimeException("Cannot find the driver in the classpath!", e);
019                }
020           
021           //Initialize Variables
022           String url = "jdbc:mysql://localhost:3306/javabase";
023           String username = "java";
024           String password = "password";
025           String query = null;
026           String update = null;
027           java.sql.Statement stmnt= null;
028           java.sql.Connection connection = null;
029           Scanner in = new Scanner(System.in);
030           
031           try {
032               System.out.println("Connecting database...");
033               connection = DriverManager.getConnection(url, username, password);
034               System.out.println("Database connected!");
035               stmnt = connection.createStatement();
036               
037               boolean done = false;
038               boolean queryDone = false;
039               boolean updateDone = true;
040               while(!done)
041               {
042                   while(!queryDone){
043                           System.out.println("Please return query. Return \"update\" if you wish to update. Return \"exit\" if you wish to exit.");
044                           query = in.nextLine();
045                               if(query.equalsIgnoreCase("exit"))
046                               {
047                                   updateDone = true;
048                                   queryDone = true;
049                                   done = true;
050                                   break;
051                               }
052                               if(query.equalsIgnoreCase("update"))
053                               {
054                                   updateDone = false;
055                                   queryDone = true;
056                                   done = false;
057                                   break;
058                               }
059                               try{
060                                ResultSet rs = stmnt.executeQuery(query);
061                                ResultSetMetaData metaData = rs.getMetaData();
062                                int count = metaData.getColumnCount();
063
064                                while(rs.next()){
065                                        for (int i = 1; i <= count; i++)
066                                        {
067                                            if(i>1)
068                                                System.out.print(", ");
069                                            String columnValue = rs.getString(i);
070                                            String columnName = metaData.getColumnName(i);
071                                            System.out.print(columnName+": "+ columnValue);
072                                        }
073                                        System.out.println("");
074                                }
075                                
076                                System.out.println("Query successful.");
077                               }
078                               catch(SQLException e)
079                               {
080                                   System.out.println("Not a proper query!");
081                                   throw new Exception("Statement Input error!",e);
082                               }
083                   }
084                   while(!updateDone){
085                           System.out.println("Please return an update. Return \"query\" if you wish to insert a query. Return \"exit\" if you wish to exit.");
086                           update = in.nextLine();
087                               if(update.equalsIgnoreCase("exit"))
088                               {
089                                   updateDone = true;
090                                   queryDone = true;
091                                   done = true;
092                                   break;
093                               }
094                               if(update.equalsIgnoreCase("query"))
095                               {
096                                   updateDone = true;
097                                   queryDone = false;
098                                   done = false;
099                                   break;
100                               }
101                               try{
102                                stmnt.executeUpdate(update);
103                                System.out.println("Update successful.");
104                               }
105                               catch(SQLException e)
106                               {
107                                   System.out.println("Not a proper update!");
108                                   throw new Exception("Statement Input error!",e);
109                               }
110                   }
111                       
112               }
113           } catch (SQLException e) {
114               throw new RuntimeException("Cannot connect the database!", e);
115           } finally {
116               System.out.println("Closing the connection.");
117               if (stmnt != null)
118                   try { 
119                           connection.close(); 
120                                System.out.println("Goodbye");
121                                } 
122               catch (SQLException ignore) {}
123           }
124   }
125}
